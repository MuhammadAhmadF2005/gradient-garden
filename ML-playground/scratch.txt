    "def best_split(X, y):\n",
    "    # try every feature and threshold, return best (feature, threshold)\n",
    "    y = y.ravel()\n",
    "    best_ig, best_feat, best_thresh = -1, None, None\n",
    "    for feat in range(X.shape[1]):\n",
    "        thresholds = np.unique(X[:, feat])\n",

    "        \"threshold\": thresh,\n",

    "    }\n",
